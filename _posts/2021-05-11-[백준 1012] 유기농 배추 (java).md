package basics_600;

import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.LinkedList;
import java.util.Queue;
import java.util.StringTokenizer;

public class BOJ1012 {
	static int[][] board;
	static boolean[][] visited;
	static int M, N, K;
	static int[] dx = { -1, 0, 1, 0 };
	static int[] dy = { 0, 1, 0, -1 };

	public static void main(String[] args) throws IOException {
		// TODO Auto-generated method stub
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));

		// tc입력
		int tc = Integer.parseInt(br.readLine());

		// tc만큼 반복
		while (tc-- > 0) {

			// 초기화
			StringTokenizer st = new StringTokenizer(br.readLine());
			M = Integer.parseInt(st.nextToken());
			N = Integer.parseInt(st.nextToken());
			K = Integer.parseInt(st.nextToken());

			board = new int[M][N];
			visited = new boolean[M][N];
			int count = 0;

			// 배추 위치 표시
			for (int i = 0; i < K; i++) {
				st = new StringTokenizer(br.readLine());
				int m = Integer.parseInt(st.nextToken());
				int n = Integer.parseInt(st.nextToken());
				board[m][n] = 1;
			}

			for (int i = 0; i < M; i++) {
				for (int j = 0; j < N; j++) {
					if (!visited[i][j] && board[i][j] == 1) { // 탐색한적이 없고, 배추가 심어져 있다
						bfs(i, j);
						count++; // 새로운 영역을 발견한거니까 ++
					}
				}
			}
			System.out.println(count);

		}

	}

	static void bfs(int x, int y) {
		Queue<Integer> queueX = new LinkedList<Integer>();
		Queue<Integer> queueY = new LinkedList<Integer>();

		visited[x][y] = true; // 방문했으니까 trrue
		queueX.offer(x); // 탐색해야하니까 큐에 넣어줌
		queueY.offer(y);

		while (!queueX.isEmpty()) {
			int tempX = queueX.poll(); // 큐에서 꺼내 탐색 시작
			int tempY = queueY.poll();

			for (int i = 0; i < 4; i++) {

				int moveX = tempX + dx[i]; // 상,우,하,좌 순으로 탐색
				int moveY = tempY + dy[i];

				if (moveX >= 0 && moveY >= 0 && moveX < M && moveY < N // 해당 범위 내에 있고
						&& !visited[moveX][moveY] && board[moveX][moveY] == 1) { // 방문한적 없으며 배추가 심어져있는 곳이라면
					visited[moveX][moveY] = true; // 방문하고
					queueX.offer(moveX); // 큐에 넣어준다.
					queueY.offer(moveY);

				}

			}

		}

	}

}
